## ❓ Question:

> Given an integer array `nums` that may contain duplicates, return _all possible_ _subsets_ _(the power set)_.

The solution set **must not** contain duplicate subsets. Return the solution in **any order**.

**Example 1:**

**Input:** nums = [1,2,2]
**Output:** [[],[1],[1,2],[1,2,2],[2],[2,2]]

**Example 2:**

**Input:** nums = [0]
**Output:** [[],[0]]e

class Solution {

public:

    vector<vector<int>> subsetsWithDup(vector<int>& nums) {

        vector<vector<int>> result;

        vector<int> buff;

        sort(nums.begin(), nums.end());  // IMPORTANT: needed for duplicate handling

        solve(nums, 0, buff, result);

        return result;

    }

    private:

    void solve(vector<int>& nums, int index, vector<int>& buff, vector<vector<int>>& result) {

        // Every path is a valid subset (including empty)

        result.push_back(buff);

        // Try adding each remaining element

        for(int i = index; i < nums.size(); i++) {

            // Skip duplicates: if same as previous AND not the first choice at this level

            if(i != index && nums[i] == nums[i-1]) {

                continue;

            }

            buff.push_back(nums[i]);       // Choose

            solve(nums, i + 1, buff, result);  // Explore

            buff.pop_back();               // Un-choose

        }

        // No explicit return needed - function ends when loop finishes

    }

};

class Solution {
public:
    vector<vector<int>> subsetsWithDup(vector<int>& nums) {
        vector<vector<int>> result;
        vector<int> buff;
        
        sort(nums.begin(), nums.end());
        solve(nums, 0, buff, result);
        return result;
    }
    
private:
    void solve(vector<int>& nums, int index, vector<int>& buff, vector<vector<int>>& result) {
        // Base case: processed all elements
        if(index == nums.size()) {
            result.push_back(buff);
            return;
        }
        
        // Choice 1: Include nums[index]
        buff.push_back(nums[index]);
        solve(nums, index + 1, buff, result);
        buff.pop_back();
        
        // Choice 2: Exclude nums[index] AND all its duplicates
        while(index + 1 < nums.size() && nums[index] == nums[index + 1]) {
            index++;
        }
        solve(nums, index + 1, buff, result);
    }
};
